ORG 0x2BF

CURRENT_LEN:	WORD 0x0
PTR:	WORD 0x5FD

START:
	CLA

GET_LEN:
	IN 7
	AND #0x40
	BEQ GET_LEN
	IN 6
	ST 0x5FD
	ST CURRENT_LEN
	CMP (PTR)+
	CLA
	ST (PTR)

CLEAR:
	CLA

ADD_SYMBOL:
	IN 7
	AND #0x40
	BEQ ADD_SYMBOL
	LD (PTR)
	BNE IS_SECOND
	IN 6
	SWAB
	ST (PTR)
	JUMP NEXT

IS_SECOND:
	IN 6
	ST (PTR)
	CMP (PTR)+
	CLA
	ST (PTR)

NEXT:
	LOOP CURRENT_LEN
	JUMP CLEAR
	HLT

ORG 0x5FD
STRING: WORD 0
